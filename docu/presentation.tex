if_statement

% ----- AST -----

fonksion triinsertion(liste)
    he n = liste
    he i = 1 
    w (i < n)
        he element = n[i]
        he j = i - 1       
        w (j >= 0 e n[j] > element)
            n[j + 1] = n[j]
            j = j - 1
        mi  
        n[j + 1] = element
        i = i + 1
    mi 
    tem n
mi

he nombres = bi[5, 2, 8, 1, 3]
